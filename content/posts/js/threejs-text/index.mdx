## 加载字体

```js
// 通过fontLoader加载字体json文件
fontLoader.load("/fonts/helvetiker_regular.typeface.json", function (font) {
  // 创建几何体
  const textGeometry = new TextGeometry("PAGGY", {
    // 字体
    font,
    // 大小
    size: 0.5,
    // 深度
    height: 0.2,
    // 细分
    curveSegments: 12,
    // 倒角
    bevelEnabled: true,
    // 倒角
    bevelThickness: 0.03,
    // 倒角大小
    bevelSize: 0.02,
    // 倒角位置
    bevelOffset: 0,
    // 倒角细分
    bevelSegments: 5,
  });
  // 字体居中
  textGeometry.center();
})

```
