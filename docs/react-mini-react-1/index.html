7function mount(vnode, container) {
8  const { vtype } = vnode;
9  if (!vtype) {
10    //
11    mountTextNode(vnode, container);
12  }
13  if (vtype === 1) {
14    //
15    mountHtml(vnode, container);
16  }
17  if (vtype === 3) {
18    //
19    mountFunc(vnode, container);
20  }
21  if (vtype === 2) {
22    //class
23    mountClass(vnode, container);
24  }
25}
26
27//
28function mountTextNode(vnode, container) {
29  const node = document.createTextNode(vnode);
30  container.appendChild(node);
31}
32
33//
34function mountHtml(vnode, container) {
35  const { type, props } = vnode;
36  const node = document.createElement(type); //dom
37  const { children, ...rest } = props;
38  children.map((item) => {
39    //
40    if (Array.isArray(item)) {
41      item.map((c) => {
42        mount(c, node);
43      });
44    } else {
45      mount(item, node);
46    }
47  });
48  Object.keys(rest).map((item) => {
49    if (item === "className") {
50      node.setAttribute("class", rest[item]);
51    }
52    if (item.slice(0, 2) === "on") {
53      node.addEventListener("click", rest[item]);
54    }
55  });
56  container.appendChild(node);
57}
